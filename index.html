// --- GERAR PDF MENSAL / SEMANAL DEDICADO (CORRIGIDO) --- async function exportarPDF() { const isAdmin = usuarioLogado.role === 'admin'; const tituloPDF = `Agenda Semanal (${datasDaSemanaAtual[0].dataFormatada} - ${datasDaSemanaAtual[6].dataFormatada})`; const subTitulo = isAdmin ? 'Relatório Geral (Todos os Colaboradores)' : `Colaborador: ${usuarioLogado.user}`; // Cria o elemento container temporário const element = document.createElement('div'); element.className = 'pdf-export-container'; // IMPORTANTE: Deixamos o elemento fixo na tela porém invisível para o usuário, // garantindo que o html2canvas consiga calcular largura, altura e estilos CSS. element.style.position = 'fixed'; element.style.left = '0'; element.style.top = '0'; element.style.width = '1000px'; element.style.zIndex = '-9999'; element.style.opacity = '1'; let gridDiasHTML = ''; datasDaSemanaAtual.forEach(diaInfo => { const eventosDoDia = agendamentos.filter(ev => { const mesmaData = ev.data === diaInfo.dataISO; return isAdmin ? mesmaData : (mesmaData && ev.usuario.toLowerCase() === usuarioLogado.user.toLowerCase()); }); eventosDoDia.sort((a, b) => a.hora.localeCompare(b.hora)); let evHTML = ''; eventosDoDia.forEach(ev => { evHTML += `
${ev.hora} - ${ev.local}
${ev.desc ? `${ev.desc}
` : ''} ${isAdmin ? `[${ev.usuario}]` : ''}
`; }); gridDiasHTML += `
${diaInfo.nomeDia.split('-')[0]}
${diaInfo.dataFormatada}
${evHTML || '

Sem tarefas

'}
`; }); element.innerHTML = `

📅 ${tituloPDF}

${subTitulo}

${gridDiasHTML}
`; document.body.appendChild(element); // Aguarda 300ms para o navegador renderizar completamente o HTML na tela await new Promise(resolve => setTimeout(resolve, 300)); // Configuração do HTML2PDF ajustada const opt = { margin: 8, filename: `Agenda_${usuarioLogado.user}_${datasDaSemanaAtual[0].dataISO}.pdf`, image: { type: 'jpeg', quality: 0.98 }, html2canvas: { scale: 2, useCORS: true, logging: false }, jsPDF: { unit: 'mm', format: 'a4', orientation: 'landscape' } }; try { await html2pdf().set(opt).from(element).save(); } catch (err) { alert("Erro ao gerar PDF: " + err.message); } finally { // Remove o elemento da tela após a conclusão document.body.removeChild(element); } }
